Training

metrics = []
for i in trange(5): # 5
    torch.random.manual_seed(i)
    model_class = LSTM(
        window_size=32,
        batch_size= 64,
        lr=1e-4,
        num_epochs=35,
        verbose=False,
        device='cuda'
    )
    model_class.fit(data, target)
    metrics.append(model_class.evaluate(test_data, test_target))
